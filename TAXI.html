    TAXI-trace

    • Vì đề bài chỉ có \(2 * n\) \((n \le 10) \) đỉnh và \(1\) đỉnh có thể được đi qua đi lại nhiều lần nên khi muốn đi từ \(u\) đến \(v\) thì ta sẽ đi theo đường đi ngắn nhất, do đó ta dùng floyd để có thể tạo ra mảng \(d[u][v]\) lưu đường đi ngắn nhất từ \(u\) đến \(v\) (xem thêm về floyd ở đây:VNOI - graph-theory). Gọi \(dp[i][mask]\) là cách đi ngắn nhất sao cho với mỗi bit thứ \(i (0 \le i < n) \) nếu bằng \(1\) thì ta đã đi qua đỉnh thứ \(i + 1\) và ngược lại với \(0\), với mỗi bit thứ \(i\) \((n \le i < n * 2)\) nếu bằng \(1\) thì ta đã giao hàng cho đỉnh thứ \(i + 1\) và ngược lại với \(0\), vậy nên bit \(i\) phải được bật trước bit \(i + n\) \( (0 \le i < n) \). Ta sẽ xây dựng hàm quy hoạch động như bình thường (khá dễ nên bỏ qua). Ở phần truy vết, thì ta gọi \(par[i][mask]\) là đỉnh \(j\) sao cho ở khi ở trạng thái \({i, mask}\) thì chọn đỉnh \(j\) sẽ cho ta đường đi tối ưu nhất, quá trình xây dựng mảng \(par[i][mask]\) được thực hiện cùng với việc xây dựng mảng \([dp]\). Ta sẽ đi từ trạng thái \({i, mask}\) đầu tiên (hoặc cuối cùng tuỳ thuộc vào cách dp) và dùng \(par[i][mask]\) để biết được trạng thái tiếp theo là gì. Chi tiết trong phần code.

      Độ phức tạp: \(O(2^n * n^2)\)

    Code

    
#include<bits/stdc++.h>
#define MASK(i) (1 << (i))
#define BIT(x, i) (((x) >> (i)) & 1)
#define fi first
#define se second
#define ull unsigned long long
#define ll long long
#define pii pair
#define mp make_pair
#define pb push_back
#define nl cout << "\n"
#define ___ << " " <<
#define mem(a, b) memset((a), (b), sizeof((a)))
#define all(c) (c).begin(), (c).end()
#define Times cerr << "\nTime run: " << clock() / 1000.0 << " ms\n"
#define file "test"
using namespace std;

template void mini(T1 &a, T2 b) {if (a > b) a = b;}
template void maxi(T1 &a, T2 b) {if (a < b) a = b;}

const int oo = 1e9 + 7;
const int mod = 1e9 + 7;
const int N = 2e2 + 5;
const int LOG = 20;

int n , adj[21][21] , dp[21][(1 << 20)];
int par[21][(1 << 20)] , path[21][21];

void prepare(){
    //floyd
    for(int k = 0 ; k <= 2 * n ; k++){
        for(int i = 0 ; i <= 2 * n ; i++){
            for(int j = 0 ; j <= 2 * n ; j++){
                if(adj[i][j] > adj[i][k] + adj[k][j]){
                    adj[i][j] = adj[i][k] + adj[k][j];
                    //trace cua floyd
                    path[i][j] = path[k][j];
                }
            }
        }
    }
}

int sol(int i , int mask){	
    if(mask == (1 << (2 * n)) - 1) return adj[i][0];

    if(dp[i][mask] != -1) return dp[i][mask];

    int res = oo;

    for(int j = 1 ; j <= 2 * n ; j++){
        // nếu đỉnh j đã thăm hay đã giao thì bỏ qua
        if(BIT(mask , j - 1)) continue;

        if(j > n){
            if(BIT(mask , j - n - 1)){
                int tmp = sol(j , mask ^ (1 << (j - 1))) + adj[i][j];
                if(res > tmp){
                    res = tmp;
                    par[i][mask] = j;
                    //lưu lại cách chọn đỉnh tiếp theo tối ưu
                }
            }
        }
        else{
            int tmp = sol(j , mask ^ (1 << (j - 1))) + adj[i][j];
            if(res > tmp){
                res = tmp;
                par[i][mask] = j;
                //lưu lại cách chọn đỉnh tiếp theo tối ưu
            }
        }
    }

    return dp[i][mask] = res;
}

void inp()
{
    cin >> n;
    for(int i = 0 ; i <= 2 * n ; i++){
        for(int j = 0 ; j <= 2 * n ; j++){
            cin >> adj[i][j];
            path[i][j] = i;
        }
    }
}

void solve()
{
    prepare();
    mem(par , -1);
    mem(dp , -1);

    cout << sol(0 , 0) << '\n';
    
    // truy vết đường đi

    queue  trace;

    // trạng thái đầu tiên
    int i = 0 , mask = 0;

    while(i != -1){
        trace.push(i);

        if(par[i][mask] != -1){
            stack  st;

            int nxt_i = path[i][par[i][mask]];
            
            // lấy đường đi từ i đến nxt_i tìm được bằng floyd

            while(nxt_i != i){
                st.push(nxt_i);
                nxt_i = path[i][nxt_i];
            }
            
            // lưu trữ đường đi vào biến kết quả

            while(st.size()){
                trace.push(st.top());
                st.pop();
            }
        }
        else{
            int tmp = 0;

            stack  st;

            while(tmp != i){
                st.push(tmp);
                tmp = path[i][tmp];
            }

            while(st.size()){
                trace.push(st.top());
                st.pop();
            }
        }
        
        // chuyển {i, mask} sang trạng thái tiếp theo

        i = par[i][mask];

        if(i != -1) mask ^= (1 << (i - 1));
    }

    cout << trace.size() << '\n';

    while(trace.size()){
        cout << trace.front() << ' ';
        trace.pop();
    }
}		

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    inp();
    solve();
}
